1 /** Label configurations for nodes.
  2  */
  3 
  4 $.extend(Config, {
  5 		//Label styling is done via CSS!
  6 		Label:
  7 		{
  8 			$extend: true,
  9 			type: 'HTML',
 10 			overridable: true,
 11 		},
 12 		onCreateLabel: function(domElement, node)
 13 		{
 14 			if(node.name)
 15 				domElement.innerHTML = node.name.substr(0, 10);
 16 		},
 17 		//Change some label dom properties.
 18 		//This method is called each time a label is plotted.
 19 		onPlaceLabel: function(domElement, node)
 20 		{
 21 			var style = domElement.style;
 22 			style.display = '';
 23 			style.cursor = 'pointer';
 24 
 25 		// Positions the label a bit nicely than the default
 26 			var left = parseInt(style.left);
 27 			var w = domElement.offsetWidth;
 28 			style.left = (left - w / 2) + 'px';
 29 		}		
 30 });
