  6 Config.Tips = 
  7 {
  8 			enable: true,
  9 			type: 'Native',
 10 			align: 'left',
 11 			
 12 			onShow: function(tip, node)
 13 			{
 14 				if(ctxMenuOpen)
 15 					return false;
 16 				tip.innerHTML = "";
 17 				if (!node) return false;
 18 				//edge data
 19 				if(node.nodeFrom)
 20 				{
 21 						tip.innerHTML += '<b>' + node.nodeFrom.id + ' <-> ' + node.nodeTo.id + '</b><br/>';
 22 						tip.innerHTML += "bitscore: " + node.data.bitscore + "<br/>";
 23 						tip.innerHTML += "e-value: " + node.data.error_value + "<br/>";
 24 				}
 25 				//node data
 26 				else if(node.data.type != 'enzyme')
 27 				{
 28 					//it's a read or db entry
 29 					var source = node.data.source || node.data.sample;
 30 					tip.innerHTML += "<b>" + node.id + "<br/>(" + source + ")</b><br/>";
 31 					tip.innerHTML += node.data.description + "<br/>";
 32 				}
 33 				else
 34 				{
 35 					tip.innerHTML = "<b>" + node.id + "</b><br/>" + node.data.name + "<br/>";
 36 				}
 37 				if(node.data.hidden_nodes_count) tip.innerHTML += "<b>Matching hidden nodes:</b> " + node.data.hidden_nodes_count + "<br/>";
 38 			}
 39 };
