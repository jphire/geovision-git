1 /** Bitscore filtering & coloring related stuff */
  2 /** http://mjijackson.com/2008/02/rgb-to-hsl-and-rgb-to-hsv-color-model-conversion-algorithms-in-javascript
  3  * Converts an HSV color value to RGB. Conversion formula
  4  * adapted from http://en.wikipedia.org/wiki/HSV_color_space.
  5  * Assumes h, s, and v are contained in the set [0, 1] and
  6  * returns r, g, and b in the set [0, 255].
  7  *
  8  * @param   Number  h       The hue
  9  * @param   Number  s       The saturation
 10  * @param   Number  v       The value
 11  * @return  Array           The RGB representation
 12  */
 13 function hsvToRgb(h, s, v){
 14     var r, g, b;
 15 
 16     var i = Math.floor(h * 6);
 17     var f = h * 6 - i;
 18     var p = v * (1 - s);
 19     var q = v * (1 - f * s);
 20     var t = v * (1 - (1 - f) * s);
 21 
 22     switch(i % 6){
 23         case 0: r = v, g = t, b = p; break;
 24         case 1: r = q, g = v, b = p; break;
 25         case 2: r = p, g = v, b = t; break;
 26         case 3: r = p, g = q, b = v; break;
 27         case 4: r = t, g = p, b = v; break;
 28         case 5: r = v, g = p, b = q; break;
 29     }
 30 
 31     return [r * 255, g * 255, b * 255];
 32 }
 33 var bitscoreColorMin, bitscoreColorMax;
 34 
 35 /** Calculates colors for all adges and nodes in the graph and add that information
 36  * in the nodes and edges. Color can be accessed through node.data.$color.
 37  */
 38 function colorEdges(){
 39 	
 40 	var min = Math.min, max = Math.max;
 41 	var maxScore = 0;
 42 	var minScore = 100000;
 43 	rgraph.graph.eachNode(function(node) {
 44 		var nodeMaxScore = 0;
 45 		var nodeMinScore = 100000;
 46 		node.eachAdjacency(function(adj) {
 47 			var bs = adj.data.bitscore;
 48 			if(!bs) return;
 49 				maxScore = max(bs, maxScore);
 50 				minScore = min(bs, minScore);
 51 				nodeMaxScore = max(bs, nodeMaxScore);
 52 				nodeMinScore = min(bs, nodeMinScore);
 53 		});
 54 		node.data.bitscore = nodeMaxScore;
 55 		node.data.min_bitscore = nodeMinScore;
 56 	});
 57 	if(bitscoreColorMin)
 58 	{
 59 		minScore = bitscoreColorMin;
 60 		maxScore = bitscoreColorMax;
 61 	}
 62 	/** Calculates colors based on the bitscore given as argument. Returns the
 63 	 * color in hexadecimal format.
 64 	 */
 65 	function color(bitscore){
 66 		
 67 		if(bitscoreColorMin)
 68 		{
 69 			if(bitscore > bitscoreColorMax)
 70 				return '#00ff00';
 71 			if(bitscore < bitscoreColorMin)
 72 				return '#ff0000';
 73 		}
 74 		// Use the HSV color model for interpolating between colors. Red is 0 degrees and green 120 degrees == 1/3: of a circle
 75 		var hue = (minScore == maxScore) ? 1.0/3.0 : (1.0/3.0 * (bitscore - minScore) / (maxScore - minScore));
 76 		var rgb = hsvToRgb(hue, 1.0, 1.0);
 77 		return $jit.util.rgbToHex([Math.round(rgb[0]), Math.round(rgb[1]), Math.round(rgb[2])]);
 78 	}
 79 	rgraph.graph.eachNode(function(node) {
 80 		node.eachAdjacency(function(adj) {
 81 			adj.data.$color = color(adj.data.bitscore);
 82 		});
 83 		if(node.data.type == 'enzyme')
 84 			node.data.color = color(node.data.bitscore);
 85 	});
 86 	return false;
 87 }
 88 /** Function to filter graph based on a bitscore given by the user.
 89  */
 90 function filter(bitscore) {
 91 	if (isNaN(bitscore) || bitscore <= 0) { /*bitscores must make sense*/
 92 		$('#filtererror').html("Not a valid bitscore.<br/>");
 93 		return false;
 94 	}
 95 	saveUndoState();
 96 	rgraph.op.deleteUntagged(bitscore);
 97 return true;
 98 }
