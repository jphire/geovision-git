1 /**
  2  * Function for showing the alignment of the read and the db-entry and coloring it so that you can easily see which parts match
  3  * @param thisid the id of the edge for which the alignment is shown
  4  * @param nodeFrom the id of the node on the other side of the edge, only used for showing to the user so it's easier to distinguish alignments from eachother
  5  * @param nodeTo the id of the node on the other side of the edge, only used for showing to the user so it's easier to distinguish alignments from eachother
  6  */
  7 function alignmentfunction(thisid, nodeFrom, nodeTo) {
  8 	$.getJSON('/show_alignment', {id: thisid}, function (data) { /*get the json with the data*/
  9 		if(data == null){
 10 			return false;
 11 		}
 12 		alignment = $('<div class="alignment"></div>');
 13 		var close = $('<div id = "closealign">Close</div>');
 14 		close.appendTo(alignment);
 15 		$('<p>Alignment for edge: <b>' + nodeFrom + ' <-> ' + nodeTo + '</b></p>').appendTo(alignment);
 16 		for ( i = 0; i < data.readseq.length; i++){
 17 			var alignmentclass = "";
 18 			if (data.readseq.charAt(i) === data.dbseq.charAt(i)){
 19 				alignmentclass = " alignsimilarity";
 20 			}
 21 			else {
 22 				alignmentclass="";
 23 			}
 24 			var a = $('<span class="alignmentpart'+ alignmentclass +'">' + data.readseq.charAt(i) +'\n'+ data.dbseq.charAt(i)+ '</span>');
 25 			a.appendTo(alignment);
 26 		}
 27 		$('#alignment').after(alignment);
 28 		alignment.slideDown(300, function() {
 29 							alignment.css('margin-bottom', '10px');
 30 		});
 31 		return true;
 32 	});
 33 	return false;
 34 }
 35 /**when the close button appears, it's set to work*/
 36 $('#closealign').live('click', function() {
 37 	closealignment(this);
 38 });

 40 /**
 41  * Function to close and delete the div-element showing the alignment
 42  * @button the DOM-element which was clicked to start this function
 43  */
 44 function closealignment (button) {
 45 		button = $(button);
 46 		var alignment = button.parent();
 47 		$(alignment).find('*').remove();/*!Hide all elements*/
 48 		$(button).remove();
 49 		$(alignment).css('margin-bottom', '0px');
 50 		$(alignment).slideUp();
 51 }

 53 /** Function to list all names, reactions and pathways related to an enzyme in the right container.
 54  * @param json the json to get info from
 55  */
 56 function showEnzymeData (json){
 57 	enzymes = {};
 58 	rgraph.graph.eachNode(function(n) {
 59 		if(n.data.type == 'enzyme')
 60 			enzymes[n.id] = n;
 61 	});
 62 	ec = json.id;
 63 
 64 	var html = '<br/>';
 65 	// KEGG pathway url coloring params: pwnumber / ecnumber <TAB> #bgcol,#fgcol / ecnumber <TAB> #bgcol,fgcol ....
 66 	if(json.pathways)
 67 	{
 68 		html += '<strong>Pathways of ' + ec + '</strong><br/>';
 69 		html += $.map(json.pathways, function(pw){
 70 			var pathwayEnzymes = $.grep(pw.enzymes, function(x) { return enzymes[x]; });
 71 			var colorUrl = escape($.map(pathwayEnzymes, function(ec) { return ec + "\t" + enzymes[ec].data.color + ',#000000'; }).join('/'));
 72 			return  '<a target="_blank" href="http://www.genome.jp/kegg-bin/show_pathway?map' + pw.id + '/' + colorUrl + '">' + pw.id + ': ' + pw.name + '</a><br/>';
 73 		}).join('');
 74 	}
 75 
 76 	names = json.names;
 77 
 78 	html += '<strong>Other names of ' + ec + ':</strong><br/>';
 79 	for (name in names){
 80 		html = html + names[name] + '<br/>';
 81 	}
 82 	//#names is replaced fully when this is called for a new node
 83 	$('#names').html(html); 
 84 	return;
 85  }
