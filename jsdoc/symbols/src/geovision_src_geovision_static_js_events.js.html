1 /** Event configurations for nodes and edges is implemented here. onClick event is
  2  * triggered when a node or an edge is clicked. 
  3  *
  4  * When node is is clicked, a new database query is made to fetch new adjacencies for 
  5  * the clicked node. fetchJSON also calls JIT graph.op.sum to sum the graphs and animate
  6  * the transition.
  7  *
  8  * For enzymes additional data is fetched using enzyme_data function in
  9  * [project root]/src/geovision/viz/views.py and shown with showEnzymeData in moreinfo.js.
 10  *
 11  * If an edge is clicked, nothing happens. The right-click functionality is implemented in 
 12  * contextmenu.js. 
 13  *
 14  * The onMouseEnter and onMouseLeave events change the mouse pointer style as well 
 15  * as some other values related to all events.
 16  */
 17 
 18 Config.Events = 
 19 {
 20 	enableForEdges: true,
 21 	enable : true,
 22 	type : 'Native', //edge event doesn't work with 'HTML'
 23 
 24 	onClick: function(node, opt){
 25 		if(!node || node.nodeFrom)
 26 			return;
 27 		if(currentEdge != undefined){
 28 			rgraph.config.Events.onMouseLeave(currentEdge);
 29 		}
 30 		if(currentNode != undefined){
 31 			rgraph.config.Events.onMouseLeave(currentNode);
 32 		}
 33 		saveUndoState();
 34 
 35 		// Fetch new adjacencies for the clicked node, sum and animate
 36 		fetchJSON(node, true);
 37 
 38 		var changed_max = false;
 39 		if(node._depth >= max_level){
 40 			max_level = node._depth + 2;
 41 			changed_max = true;
 42 		}
 43 
 44 		// Increase the number of concentric circles if needed
 45 		if(changed_max){
 46 			$jit.Graph.Util.computeLevels(rgraph.graph, rgraph.root, 0);
 47 			rgraph.canvas.canvases[1].opt.numberOfCircles = max_level;
 48 			rgraph.canvas.canvases[1].plot();
 49 			changed_max = false;
 50 		}
 51 
 52 
 53 		if(node.data.type == 'enzyme'){
 54 			$.getJSON('/enzyme_data', { id: node.id }, showEnzymeData);
 55 		}
 56 		else{
 57 			var html = '';
 58 			var source = node.data.source || node.data.sample;
 59 			html += "<b>" + node.id + "<br/>(" + source + ")</b><br/>";
 60 			html += node.data.description + "<br/>";
 61 			if(node.data.source == 'uniprot'){
 62 				html += node.data.sub_db + "<br/>";
 63 				html += node.data.entry_name + "<br/>";
 64 				html += node.data.os_field + "<br/>";
 65 				html += node.data.other_info + "<br/>";
 66 			}
 67 
 68 			$('#names').html(html);
 69 		}
 70 	},
 71 	onMouseEnter: function(node, eventInfo, e) {
 72 		if(ctxMenuOpen || busy)
 73 			return;
 74 
 75 		rgraph.canvas.getElement().style.cursor = 'pointer';
 76 
 77 		if (node.nodeTo)
 78 			currentEdge = node;
 79 		else
 80 			currentNode = node;
 81 	},
 82 	onMouseLeave: function(object, eventInfo, e) {
 83 		if(ctxMenuOpen || !object || busy)
 84 			return;
 85 		currentNode = currentEdge = undefined;
 86 		rgraph.canvas.getElement().style.cursor = '';
 87 	}
 88 };
