1 var currentNode;
  2 var currentEdge;
  3 var ctxMenuOpen;
  4 
  5 /** Called when the context menu is being hidden */
  6 function hideCtxMenu()
  7 {
  8 	if(!ctxMenuOpen) return;
  9 	ctxMenuOpen = false;
 10 	if(currentNode) rgraph.config.Events.onMouseLeave(currentNode)
 11 	if(currentEdge) rgraph.config.Events.onMouseLeave(currentEdge);
 12 	busy = false;
 13 	currentEdge = currentNode = false;
 14 	rgraph.config.Navigation.panning = true;
 15 	rgraph.config.Tips.enable = true;
 16 	rgraph.events.pressed = false;
 17 }
 18 /** Adds the context menu to the canvas and binds all the necessary events.
 19 * The event bindings are done in the jquery-contextmenu plugin code */
 20 function initContextMenu()
 21 {
 22 	$('#infovis').contextMenu('nodeMenu', {
 23 		'shadow': false,
 24 		'bindings': {
 25 			/* The actual menu events. All the menu entries are defined in templates/graphviz.html */
 26 			'close': function() { },
 27 			'n_center': function() {
 28 				console.log(currentNode.id);
 29 				var id = currentNode.id;
 30 				setBusy('Centering');
 31 				tagNode(rgraph.graph.getNode(id), false);
 32 				rgraph.onClick(id, rgraph.op.userOptions);
 33 			},
 34 			'e_align': function() { alignmentfunction(currentEdge.data.blast_id, currentEdge.nodeFrom.id, currentEdge.nodeTo.id); },
 35 			'n_tag': function() { 
 36 				if (currentNode.traversalTag != true) {
 37 					tagNode(currentNode);
 38 				}
 39 				else {
 40 					untagNode(currentNode);
 41 				}
 42 			},
 43 			'n_more': function() { fetchJSON(currentNode); },
 44 			'n_tagsubnodes': function() { rgraph.op.tagSubnodes(currentNode)},
 45 			'n_tagsubgraph': function() { rgraph.op.tagSubgraph(currentNode)},
 46 			'n_untagsubgraph': function() { untagSubgraph(currentNode)},
 47 			'n_en_names': function() { $.getJSON('/enzyme_data', { id: currentNode.id }, showEnzymeData); },
 48 			'n_en_brendalink': function() { window.open('http://www.brenda-enzymes.org/php/result_flat.php4?ecno=' + currentNode.id); },
 49 			'n_en_kegglink': function() { window.open('http://www.genome.jp/dbget-bin/www_bget?ec:' + currentNode.id); },
 50 			'n_db_uni_link': function() { window.open('http://www.uniprot.org/uniprot/' + currentNode.id); },
 51 			'n_db_frn_link': function() { window.open('http://www.ncrna.org/frnadb/detail.html?i_name=' + currentNode.id); }
 52 
 53 		},
 54 		'onContextMenu': function(event)
 55 		{
 56 			return (currentNode || currentEdge) && !busy;
 57 		},
 58 		'onShowMenu': function(evt, menu)
 59 		{
 60 			/* Removes all the menu items that are not applicable for the currently selected node/edge */
 61 			/* The applicable menu entries are based on the prefix of the element id:
 62 			 *     e_ - for read-dbentry-edges
 63 			 *     n_ - for all nodes
 64 			 *     n_en_ - for all enzyme nodes
 65 			 *     n_db_ - for all database nodes
 66 			 *     n_db_uni - for uniprot database nodes
 67 			 *     n_db_frn - for frnadb database nodes
 68 			 *     n_db_silva - for silva database nodes
 69 			 */
 70 			ctxMenuOpen = true;
 71 
 72 			// menu clicks don't play too nicely with JIT - disable panning & tooltips temporarily and re-enable them when hiding the menu.
 73 			rgraph.config.Navigation.panning = false;
 74 			rgraph.config.Tips.enable = false;
 75 			rgraph.tips.hide();
 76 
 77 			if(!currentEdge || !currentEdge.data.blast_id ) 
 78 				$('li[id^=e_]', menu).remove();
 79 			if(!currentNode)
 80 				$('li[id^=n_]', menu).remove();
 81 			else
 82 			{
 83 				if(currentNode.data.type != 'enzyme')
 84 					$('li[id^=n_en_]', menu).remove();
 85 				if(currentNode.data.type != 'dbentry')
 86 					$('li[id^=n_db_]', menu).remove();
 87 				else
 88 				{
 89 					if(currentNode.data.source != 'uniprot')
 90 						$('li[id^=n_db_uni]', menu).remove();
 91 					if(currentNode.data.source != 'frnadb')
 92 						$('li[id^=n_db_frn]', menu).remove();
 93                     if(currentNode.data.source.indexOf('silva') == -1)
 94                     	$('li[id^=n_db_silva]', menu).remove();
 95 
 96 				}
 97 			}
 98 			return menu;
 99 		},
100 		'onHideMenu': hideCtxMenu
101 	});
102 }
