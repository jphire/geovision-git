1 /** prepareJSON(jitJSON json) - prepare a JIT graph in JSON 'graph' form for display
  2  * @param json the json to be used for the graph
  3  * @return the 'graphified' json */
  4 function prepareJSON(json)
  5 {
  6 	for (i in json)
  7 	{
  8 		var node = json[i];
  9 		var data = node.data;
 10 		if(data.type == "enzyme")
 11 			data.$color = '#0000FF';
 12 		else if (data.type == "dbentry")
 13 			data.$color = '#00FF00';
 14 		for(var adj in node.adjacencies)
 15 		{
 16 				// setting adj.$direction is required for making arrow tips point to the correct direction
 17 				// Firefox actually gets them correct completely by accident without this line.
 18 				node.adjacencies[adj].data.$direction = node.adjacencies[adj].nodeTo;
 19 		}
 20 	}
 21 	return json;
 22 }
 23 /** setBusy(null|string msg) - Set animation busy status
 24  * If msg is a string, add a loading bar with the specified message and prevent all other operations
 25  * If msg is null, disable busy state
 26  * @param msg message to be shown
 27  */
 28 function setBusy(msg)
 29 {
 30 	if(msg)
 31 	{
 32 		$('#load').html(msg + '...');
 33 		rgraph.canvas.getElement().style.cursor = 'wait';
 34 		busy = msg;
 35 	}
 36 	else
 37 	{
 38 		$('#load').html('');
 39 		rgraph.canvas.getElement().style.cursor = '';
 40 		busy = false;
 41 	}
 42 }
 43 /** clean up the graph by removing leftover stuff
 44  * (e.g. edges with $alpha 0.0) that JIT tends to leave when deleting nodes.
 45  */
 46 function cleanupGraph()
 47 {
 48 	rgraph.graph.eachNode(function(n) {
 49 		if(n.data.$alpha < 0.01)
 50 			rgraph.graph.removeNode(n.id);
 51 	});
 52 }
 53 /**
 54  * expand the graph at the specified node
 55  * @param node JitNode from which the graph needs to expand
 56  */
 57 function fetchJSON(node)
 58 {
 59 
 60 			setBusy('Expanding');
 61 			var offset = [];
 62 			node.eachAdjacency(function(adj) {
 63 				offset.push(adj.nodeTo.id);
 64 			});
 65 			var args = $jit.util.merge(query, { read: '', dbentry: '', enzyme: '', depth: 1, offset: offset});
 66 			args[node.data.type] = node.id;
 67 			$.getJSON('/graphjson', args,
 68 				function(newdata)
 69 				{
 70 					if(newdata.error_message)
 71 					{
 72 						console.log('Server error while loading JSON', newdata.error_message);
 73 						setBusy(false);
 74 						return;
 75 					}
 76 					else
 77 					{
 78 						node.data.hidden_nodes_count = newdata[0].data.hidden_nodes_count;
 79 						rgraph.op.sum(prepareJSON(newdata), rgraph.op.userOptions);
 80 					}
 81 				});
 82 }

 84 /**
 85  * This function creates the graph according to the JSON data.
 86  */
 87 function initGraph()
 88 {
 89 	$.getJSON('/graphjson', query, function(json) {
 90 		jQuery('#loader').fadeOut();
 91 		if(!json || json.error_message)
 92 		{
 93 			if(json)
 94 				$("#navierror").text(json.error_message);
 95 			openSearch();
 96 			return;
 97 		}
 98 		initContextMenu();
 99 		$('#infovis').disableSelection();
100 
101 		rgraph = new RGraph(Config);
102 		rgraph.loadJSON(prepareJSON(json), query.root || 0);
103 		$jit.Graph.Util.computeLevels(rgraph.graph, rgraph.root, 0);
104 		delete query.view_id;
105 		delete query.root;
106 
107 		colorEdges();
108 		rgraph.refresh();
109 		rgraph.op.userOptions = settings.animationsettings;
110 		rgraph.op.tagParents = tagParents;
111 		rgraph.op.tagSubgraph = tagSubgraph;
112 		rgraph.op.tagSubnodes = tagSubnodes;
113 		rgraph.op.deleteUntagged = deleteUntagged;
114 	});
115 }
