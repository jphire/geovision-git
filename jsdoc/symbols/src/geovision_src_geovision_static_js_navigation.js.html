 47 /** Function for deleting nodes from the graph.
 48  * If bitscoreLimit is not defined, all untagged nodes except root will be deleted.
 49  * If bitscoreLimit is defined, all untagged nodes (except root) without connections 
 50  * over that bitscore will be deleted. It will also preserve a path to root from each
 51  * remaining node (since many operations would break if graph became disconnected).
 52  *
 53  * Function will also clean all edges to (and from) the deleted nodes from the graph, 
 54  * and update data.bitscore field for remaining nodes.
 55  *
 56  * Will also untag all nodes once finished.
 57  * @param bitscoreLimit the limit under which nodes will be deleted
 58  */
 59 function deleteUntagged(bitscoreLimit) {
 60 	var nodesArray = [];
 61 	rgraph.graph.eachBFS(rgraph.root, function(n) { nodesArray.push(n) });
 62 	// Add nodes to end of array and pop from there to get reverse-BFS order
 63 	while (nodesArray.length > 1) {
 64 		var node = nodesArray.pop();
 65 		if (!node.traversalTag && (!bitscoreLimit || node.data.bitscore < bitscoreLimit)) {
 66 			rgraph.op.removeNode(
 67 				node.id, 
 68 				$jit.util.merge(rgraph.op.userOptions, {
 69 					onComplete: function() { cleanupGraph(); colorEdges(); updateBitscores();},
 70 					onAfterPlotNode: function(n) { n.traversalTag = false; }
 71 			}));
 72 		}
 73 		else {
 74 			tagParents(node);
 75 		}
 76 	}
 77 }

 79 /** Function for updating data.bitscore field for all nodes in the graph.
 80  * Should be used after deleting edges from the graph if relevant values are not otherwise updated.
 81  */
 82 function updateBitscores() {
 83 	rgraph.graph.eachNode(function(node) {
 84 		var bitscore = 0;
 85 		node.eachAdjacency(function(edge) {
 86 			console.log(edge);
 87 			if (edge.data.bitscore > bitscore) {
 88 				bitscore = edge.data.bitscore;
 89 			}
 90 		});
 91 		node.data.bitscore = bitscore;
 92 	});
 93 }

 95 /** Function for checking if node has a tagged path to the root node.
 96  * Assumes the node has a path to root.
 97  * @param node the node for which to check the path
 98  */
 99 function checkRootTagpath(node) {
100 	var parentNodes = node.getParents();
101 	if (parentNodes.length == 0) {
102 		if (node.id === rgraph.root) return true;
103 		else return false;
104 	}
105 	for (var i = 0; i < parentNodes.length; i++) {
106 		pnode = parentNodes[i]
107 		if (!pnode.traversalTag) continue;
108 		if (checkRootTagpath(pnode)) return true;
109 	}
110 	return false;
111 }

113 /** Function for tagging a node. If node has no tagged path to root, a path will also be tagged
114  * since otherwise deletion could result in disconnected graph.
115  * @param node the node to be tagged
116  */
117 function tagNode(node) {
118 	if (!checkRootTagpath(node)) tagParents(node);
119 	node.traversalTag = true;
120 	rgraph.refresh()
121 }

123 /** Function for tagging a path from node to root.
124  * Builds path by tagging the parent with the best (bitscore) connection to current node.
125  * Responsibility for updating the visualization is left to user, this function does
126  * not call refresh() or any animating function.
127  * @param node the node which will have its parents tagged
128  */
129 function tagParents(node) {
130 	var currentNode = node;
131 	var parents = node.getParents();
132 	while (parents.length > 0) {
133 		var bestParent = parents[0];
134 		var bestBitscore = currentNode.getAdjacency(parents[0].id).data.bitscore;
135 		for (var i = 1; i < parents.length; i++) {
136 			var tempBitscore = currentNode.getAdjacency(parents[i].id).data.bitscore;
137 			if (tempBitscore > bestBitscore) {
138 				bestParent = parents[i];
139 				bestBitscore = tempBitscore;
140 			}
141 		}
142 		bestParent.traversalTag = true;
143 		currentNode = bestParent;
144 		parents = currentNode.getParents();
145 	}
146 	node.traversalTag = true;
147 }

149 /** Tags the subnodes of the node
150  * @param node target node to have subnodes tagged*/
151 function tagSubnodes(node) {
152 	if (!checkRootTagpath(node)) tagParents(node);
153 	node.eachSubnode(function(child) {
154 		child.traversalTag = true;
155 		console.log("Child " + child.id + " tagged");
156 	});
157 	node.traversalTag = true;
158 	rgraph.refresh()
159 }

173 /** Untags a node.
174  * Since many operations rely on all nodes having a path to root, also untags all
175  * tagged nodes without a tagged path to root after untagging this node.
176  * @param node node to be untagged
177  */
178 function untagNode(node) {
179 	node.traversalTag = false;
180 	(function subn(n) {
181 		n.eachSubnode(function(ch) {
182 			ch.traversalTag = checkRootTagpath(ch);
183 			subn(ch);
184 		});
185 	})(node);
186 	rgraph.refresh()
187 }
