1 /** config.js - Initializes all globals and the config dict. Sets default
  2  * values for settings. Detailed explanation of all the configurations available
  3  * e.g. for animations can be found at the JIT documents page
  4  * http://thejit.org/static/v20/Docs/.
  5  */
  6 
  7 var rgraph;
  8 var RGraph = $jit.RGraph;
  9 var busy = false;
 10 var defaultsettings = {
 11 		animationsettings:
 12 			{duration: 1000,
 13 			transition: 'linear',
 14 			subtype: 'EaseIn',
 15 			type:"fade:con"},
 16 		settings:
 17 			{canvaswidth: 600,
 18 			canvasheight: 600}
 19 };
 20 settings = $jit.util.merge(defaultsettings, settings);
 21 
 22 /* Perform parameter validation on the user settings. Replace incorrectly formatted number fields with defaults */
 23 var numericFields = {settings: ['canvaswidth', 'canvasheight'], animationsettings: ['duration']}
 24 
 25 for (var key1 in numericFields)
 26 {
 27 	for (var i in numericFields[key1])
 28 	{
 29 		var key2 = numericFields[key1][i];
 30 		var num = parseInt(settings[key1][key2]);
 31 		settings[key1][key2] = isNaN(num) ? defaultsettings[key1][key2] : num;
 32 	}
 33 }
 34 
 35 /* Sets the transition function based on the strings in the settings dict*/
 36 settings.animationsettings.transitionname = settings.animationsettings.transition;
 37 var type = $jit.Trans[settings.animationsettings.transition];
 38 if(!type)
 39 	type = $jit.Trans[defaultsettings.animationsettings.transition]; 
 40 var subtype = type[settings.animationsettings.subtype];
 41 if(subtype)
 42 	type = subtype;
 43 settings.animationsettings.transition = type;
 44 settings.animationsettings.onComplete = function() { setBusy(false); };
 45 settings.animationsettings.onMerge = colorEdges;
 46 
 47 //set initial number of concentric circles
 48 var max_level = 6;
 49 
 50 var Config = 
 51 {
 52 		//Where to append the visualization
 53 		injectInto: 'infovis',
 54 		
 55 		//set canvas size
 56 		width: settings.settings.canvaswidth,
 57 		height:settings.settings.canvasheight,
 58 		//Optional: create a background canvas that plots
 59 		//concentric circles.
 60 		background: { numberOfCircles: max_level, CanvasStyles: { strokeStyle: '#555' } },
 61 		//set distance for nodes on different levels
 62 		levelDistance: 100,
 63 		//set transformation speed
 64 		duration: settings.animationsettings.duration,
 65 		fps: 40,
 66 		//set transformation style
 67 		transition: settings.animationsettings.transition,
 68 		//Add navigation capabilities:
 69 		//zooming by scrolling and panning.
 70 		Navigation:
 71 		{
 72 		  enable: true,
 73 		  panning: 'avoid nodes',
 74 		  zooming: 25
 75 		},
 76 		//Set Node and Edge styles.
 77 		Node:
 78 		{
 79 			overridable: true,
 80 			color: '#ff0000',
 81 			alpha: 0.6,
 82 			dim: 7.0,
 83 			angularWidth: 20,
 84 			span:1,
 85 			type: 'customCircle',
 86 			CanvasStyles: {}
 87 		},
 88 		Edge:
 89 		{
 90 			overridable: true,
 91 			color: '#888800',
 92 			alpha: 0.6,
 93 			type: 'customArrow',
 94 			lineWidth:1.5,
 95 			dim: 10,
 96 		}
 97 };
 98 /** Set correct values for the settings widgets, also bind some events */
 99 jQuery(function($) {
100 	$('#infovis').css('height', parseInt(settings.settings.canvasheight));
101 	$('#infovis').css('width', parseInt(settings.settings.canvaswidth));
102 	$('#center-container').css('width', parseInt(settings.settings.canvaswidth));
103 	$('#center-container').css('height', parseInt(settings.settings.canvasheight));
104 	$('#right-container').css('height', parseInt(settings.settings.canvasheight));
105 	$('#container').css('width', parseInt(settings.settings.canvaswidth)+400);
106 	$('#container').css('height', parseInt(settings.settings.canvasheight));
107 	$('#canvas_x').val(settings.settings.canvaswidth);
108 	$('#canvas_y').val(settings.settings.canvasheight);
109 	if (settings.animationsettings.type == 'replot'){
110 		$('#animations_off').attr('checked', 'checked');
111 	}
112 	$('#duration').val(settings.animationsettings.duration);
113 	if (settings.animationsettings.transitionname != undefined){
114 		$('#animationtype').val(settings.animationsettings.transitionname);
115 	}
116 	$('#animationsubtype').val(settings.animationsettings.subtype);
117 
118 	$('#deleteUntagged').click(function(e) { e.preventDefault(); saveUndoState(); rgraph.op.deleteUntagged(); });
119 });
