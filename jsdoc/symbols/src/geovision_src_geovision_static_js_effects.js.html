  6 /**
  7  * Opens the left side query menu.
  8  */
  9 function openSearch()
 10 {
 11 	var elem = $('#graphnavi')
 12         if (!opened){
 13             elem.find('#optiontag').hide();
 14             elem.animate({width: "25%"}, {complete:
 15 			function() { $('#navicontainer').fadeIn('fast', function() { opened = true; } ); }});
 16 	}
 17 }
 18 /**
 19  * Closes the left side query menu.
 20  */
 21 function closeSearch(e)
 22 {
 23 	if (opened){
 24 		if (!e) var e = window.event;
 25 		var tg = (window.event) ? e.srcElement : e.target;
 26 		//if (tg.nodeName != 'DIV') return;
 27 		var reltg = (e.relatedTarget) ? e.relatedTarget : e.toElement;
 28 		while (reltg != tg && reltg.nodeName != 'BODY')
 29 			reltg= reltg.parentNode
 30 		if (reltg== tg) return;
 31 		// Now we know that the mouse actually left the layer:
 32 		$('#graphnavi').find('#navicontainer').hide();/*!Hide all elements*/
 33 		$('#graphnavi').animate({width: "7px"}, {complete: function() {
 34                        $(this).find('#optiontag').fadeIn();
 35                        opened = false;
 36 		}});
 37 	}
 38 }

 64 		filter(parseFloat($('#bitscorefilter').val()));
 65 		return false;
 66 	})
 67 	$('#colorform').submit(function(){
 68 		$('#colorErrorMsg').text('');
 69 		if($('#colortypeDynamic').attr('checked'))
 70 			setBitscoreColoring();
 71 		else
 72 		{
 73 			var min = parseFloat($('#colorMin').val()), max = parseFloat($('#colorMax').val());
 74 			if(min > 0 && max > min)
 75 				setBitscoreColoring(min, max);
 76 			else
 77 				$('#colorErrorMsg').text('Enter valid numbers');
 78 		}
 79 		return false;
 80 	});

 84 	});
 85 	$('#colortypeDynamic').click(function(n){
 86 		$('#colorErrorMsg').text('');
 87 		$('#colorMin, #colorMax').attr('disabled', 'disabled');
 88 	});
