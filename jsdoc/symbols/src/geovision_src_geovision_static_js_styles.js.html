1 /** Calculate the shortest distance between a line segment and a point */
  2 dotLineLength = function( x, y, x0, y0, x1, y1, o ){
  3 	function lineLength( x, y, x0, y0 ){
  4 		return Math.sqrt( ( x -= x0 ) * x + ( y -= y0 ) * y );
  5 	}
  6 	if( o && !( o = function( x, y, x0, y0, x1, y1 ){
  7 		if( !( x1 - x0 ) ) return { x: x0, y: y };
  8 		else if( !( y1 - y0 ) ) return { x: x, y: y0 };
  9 		var left, tg = -1 / ( ( y1 - y0 ) / ( x1 - x0 ) );
 10 		return { x: left = ( x1 * ( x * tg - y + y0 ) + x0 * ( x * - tg + y - y1 ) ) / ( tg * ( x1 - x0 ) + y0 - y1 ), y: tg * left - tg * x + y };
 11 	}( x, y, x0, y0, x1, y1 ), o.x >= Math.min( x0, x1 ) && o.x <= Math.max( x0, x1 ) && o.y >= Math.min( y0, y1 ) && o.y <= Math.max( y0, y1 ) ) ){
 12 		var l1 = lineLength( x, y, x0, y0 ), l2 = lineLength( x, y, x1, y1 );
 13 		return l1 > l2 ? l2 : l1;
 14 	}
 15 	else {
 16 		var a = y0 - y1, b = x1 - x0, c = x0 * y1 - y0 * x1;
 17 		return Math.abs( a * x + b * y + c ) / Math.sqrt( a * a + b * b );
 18 	}
 19 };

 21 RGraph.Plot.NodeTypes.implement({
 22     'customCircle': {
 23       'render': function(node, canvas){
 24 		/* Circle drawing code based on JIT code
 25 		 * Adds the drawing of the traversal tag of the node
 26 		 */
 27           var pos = node.pos.getc(),
 28               radius = node.getData('dim');
 29           var ctx = canvas.getCtx();
 30           ctx.beginPath();
 31           ctx.arc(pos.x, pos.y, radius, 0, Math.PI * 2, true);
 32           ctx.closePath();
 33           ctx.fill();
 34 
 35           if(node.traversalTag)
 36           {
 37 	          ctx.fillStyle = '#ffffff';
 38 	          ctx.beginPath();
 39 	          ctx.arc(pos.x, pos.y, radius/2, 0, Math.PI * 2, true);
 40 	          ctx.closePath();
 41 	          ctx.fill();
 42           }
 43 
 44 
 45         },
 46         'contains': function(node, pos){
 47           if(node.ignore)
 48               return false;
 49           var npos = node.pos.getc(true),
 50               radius = node.getData('dim');
 51           var diffx = npos.x - pos.x,
 52               diffy = npos.y - pos.y,
 53               diff = diffx * diffx + diffy * diffy;
 54           return diff <= radius * radius;
 55         }
 56     }
 57 });

 59 RGraph.Plot.EdgeTypes.implement({  
 60 	'customArrow':{
 61 		/** Largely based on the JIT arrow rendering code, with several improvements:
 62 		 *   * doesn't draw part of the line over the arrow tip, making it look less ugly with alpha < 1
 63 		 *   * takes the node radius into account to make it look a lot nicer and easier to click a node with many connections
 64 		 *   * improved collision detection
 65 		 */
 66 	    'render': function(adj, canvas) {
 67             var from = adj.nodeFrom.pos.getc(),
 68                 to = adj.nodeTo.pos.getc(),
 69                 dim = adj.getData('dim'),
 70                 direction = adj.data.$direction,
 71 				swap = adj.nodeTo.id != direction;
 72             var ctx = canvas.getCtx();
 73             Complex = $jit.Complex;
 74             // invert edge direction
 75             if (swap) {
 76               var tmp = from;
 77               from = to;
 78               to = tmp;
 79             }
 80             var vect = new Complex(to.x - from.x, to.y - from.y);
 81 	    var norm = vect.norm();
 82             to.$add(vect.scale(-adj.nodeTo.getData("dim") / norm));
 83             from.$add(vect.$scale(adj.nodeFrom.getData("dim") / norm));
 84             vect = new Complex(to.x - from.x, to.y - from.y);
 85 
 86             vect.$scale(dim / vect.norm());
 87             var intermediatePoint = new Complex(to.x - vect.x, to.y - vect.y),
 88                 normal = new Complex(-vect.y / 2, vect.x / 2),
 89                 v1 = intermediatePoint.add(normal), 
 90                 vM = intermediatePoint.clone();
 91                 v2 = intermediatePoint.$add(normal.$scale(-1));
 92             
 93             ctx.beginPath();
 94 	    if(from.x != from.x) return;
 95             ctx.moveTo(from.x, from.y);
 96             ctx.lineTo(vM.x, vM.y);
 97             ctx.stroke();
 98             ctx.beginPath();
 99             ctx.moveTo(v1.x, v1.y);
100             ctx.lineTo(v2.x, v2.y);
101             ctx.lineTo(to.x, to.y);
102             ctx.closePath();
103             ctx.fill();
104 
105         },
106         'contains': function(adj, pos) {
107             var from = adj.nodeFrom.pos.getc(),
108 		        to = adj.nodeTo.pos.getc();
109             var lineWidth = adj.getData('epsilon');
110             var d = lineWidth/2;
111 
112             var vect = new Complex(to.x - from.x, to.y - from.y);
113 	    var norm = vect.norm();
114             to.$add(vect.scale(-adj.nodeTo.getData("dim") / norm));
115             from.$add(vect.$scale(adj.nodeFrom.getData("dim") / norm));
116 
117             var minX = Math.min(from.x, to.x) - d, maxX = Math.max(from.x, to.x) + d;
118             var minY = Math.min(from.y, to.y) - d, maxY = Math.max(from.y, to.y) + d;
119             if(pos.x < minX || pos.x > maxX || pos.y < minY || pos.y > maxY)
120 	            return false;
121         return dotLineLength(pos.x, pos.y, from.x, from.y, to.x, to.y, false) < lineWidth;
122 	    }
123     }
124 });
